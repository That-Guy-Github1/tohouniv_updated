th_kappa_counter_movement_disaster = {
	potential = {
		has_mission = KPP_enact_the_first_reform
		NOT = { mission_completed = KPP_enact_the_first_reform }
		has_reform = Gensokyan_kpp_kappa_state
		NOT = { has_country_flag = kpp_had_counter_movement_disaster }
	}
	
	can_start = {
		has_any_disaster = no
		OR = {
			custom_trigger_tooltip = {
				tooltip = kpp_enabled_easier_disasters_tt
				has_country_flag = kpp_enabled_easier_disasters
			}
			AND = {
				NOT = { stability = 0 }
				NOT = { average_home_autonomy = 10 }
			}
		}
	}
	
	can_stop = {
		custom_trigger_tooltip = {
			tooltip = kpp_not_enabled_easier_disasters_tt
			NOT = { has_country_flag = kpp_enabled_easier_disasters }
		}
		OR = {
			stability = 0
			average_home_autonomy = 20
		}
	}
	modifier = {
		global_autonomy = 0.125
		stability_cost_modifier = 0.15
		republican_tradition = -0.5
	}
		
	progress = {
		modifier = {
			factor = 2
			custom_trigger_tooltip = {
				tooltip = kpp_enabled_easier_disasters_tt
				has_country_flag = kpp_enabled_easier_disasters
			}
		}
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 2
			NOT = { average_home_autonomy = 50 }
		}
		modifier = {
			factor = 2
			NOT = { average_home_autonomy = 40 }
		}
		modifier = {
			factor = 2
			NOT = { average_home_autonomy = 30 }
		}
		modifier = {
			factor = 2
			NOT = { average_home_autonomy = 20 }
		}
		modifier = {
			factor = 2
			NOT = { average_home_autonomy = 10 }
		}
	}	
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = th_kappa_counter_movement_disaster_tt
			had_country_flag = { flag = active_kappa_counter_movement_disaster days = 3650 }
		}
		stability = 1
		NOT = { num_of_rebel_armies = 1 }
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}
	
	on_start = flavor_kpp.41
	on_end = flavor_kpp.42
	
	on_monthly = {
		events = {
		}
		
		random_events = { 
			1200 = 0
			10 = flavor_kpp.47
			10 = flavor_kpp.48
			10 = flavor_kpp.49
		}
	}
}
