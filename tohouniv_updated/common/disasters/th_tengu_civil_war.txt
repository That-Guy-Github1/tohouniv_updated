th_tengu_caste_civil_war_disaster = {
	potential = {
		has_mission = TGU_the_tengu_conflict
		NOT = { mission_completed = TGU_the_tengu_conflict }
		has_reform = Gensokyan_tgu_caste_system
		NOT = { has_country_flag = tgu_had_tengu_civil_war }
		has_estate = estate_great_tengu
	}
	
	can_start = {
		has_any_disaster = no
		OR = {
			if = {
				limit = {
					has_estate = estate_church
				}
				NOT = {
					estate_loyalty = {
						estate = estate_church
						loyalty = 30
					}
				}
			}
			if = {
				limit = {
					has_estate = estate_crow_tengu
				}
				NOT = {
					estate_loyalty = {
						estate = estate_crow_tengu
						loyalty = 30
					}
				}
			}
			if = {
				limit = {
					has_estate = estate_wolf_tengu
				}
				NOT = {
					estate_loyalty = {
						estate = estate_wolf_tengu
						loyalty = 30
					}
				}
			}
			estate_influence = {
				estate = estate_great_tengu
				influence = 80
			}
		}
		NOT = { has_estate_privilege = estate_great_tengu_promote_aya }
		NOT = { has_estate_privilege = estate_great_tengu_rights_to_the_low_castes }
		NOT = { stability = 2 }
	}
	
	can_stop = {
		has_any_disaster = yes
		OR = {
			AND = {
				if = {
					limit = {
						has_estate = estate_crow_tengu
					}
					estate_loyalty = {
						estate = estate_crow_tengu
						loyalty = 50
					}
				}
				if = {
					limit = {
						has_estate = estate_wolf_tengu
					}
					estate_loyalty = {
						estate = estate_wolf_tengu
						loyalty = 50
					}
				}
				if = {
					limit = {
						has_estate = estate_church
					}
					estate_loyalty = {
						estate = estate_church
						loyalty = 50
					}
				}
				stability = 2
			}
			has_estate_privilege = estate_great_tengu_rights_to_the_low_castes
		}
	}
	modifier = {
		stability_cost_modifier = 0.33
		technology_cost = 0.15
		reform_progress_growth = -1
		legitimacy = -1
	}
		
	progress = {
		modifier = {
			factor = 2
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 2
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 2
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 2
			NOT = { stability = -2 }
		}
		modifier = {
			factor = 2
			NOT = {
				estate_loyalty = {
					estate = estate_church
					loyalty = 40
				}
			}
		}
		modifier = {
			factor = 2
			NOT = {
				estate_loyalty = {
					estate = estate_crow_tengu
					loyalty = 40
				}
			}
		}
		modifier = {
			factor = 2
			NOT = {
				estate_loyalty = {
					estate = estate_wolf_tengu
					loyalty = 40
				}
			}
		}
		modifier = {
			factor = 2
			estate_influence = {
				estate = estate_great_tengu
				influence = 80
			}
		}
	}
	ended_by_country_breaking_to_rebels = no
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = th_tengu_caste_civil_war_disaster_ct
			OR = {
				check_variable = {
					which = th_tengu_civil_war_score
					value = 15
				}
				NOT = {
					check_variable = {
						which = th_tengu_civil_war_score
						value = -14
					}
				}
			}
		}
	}
	
	on_start = th_tengu_civil_war.1
	on_end = th_tengu_civil_war.2
	
	on_monthly = {
		events = {
		}
		
		random_events = { 
			1000 = 0
		}
	}
	on_start_effect = {
		custom_tooltip = th_tengu_caste_civil_war_disaster_tt
		add_stability = -4
		add_legitimacy = -50
	}
}
