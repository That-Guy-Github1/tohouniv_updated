#Monopolies
estate_wolf_tengu_monopoly_of_wood = {
	icon = privilege_monopoly_of_goods_slaves
	max_absolutism = -10
	loyalty = 0.1
	is_valid = {
		has_dlc = "Emperor"
		any_owned_province = {
			OR = {
				trade_goods = naval_supplies
				trade_goods = tropical_wood
			}
		}
	}
	can_select = {
		custom_trigger_tooltip = {
			tooltip = unavailable_during_revolution_disaster
			NOT = { has_disaster = revolution }
			NOT = {
				has_disaster_progress = {
					disaster = revolution
					value = 1
				}
			}
			NOT = { has_disaster = french_revolution }
			NOT = {
				has_disaster_progress = {
					disaster = french_revolution
					value = 1
				}
			}
		}
	}
	on_granted = {
		if = {
			limit = {
				OR = {
					naval_supplies = 1
					tropical_wood = 1
				}
			}
			add_mercantilism = 1
		}
		add_years_of_owned_provinces_production_income = {
			years = 8
			trigger = {
				OR = {
					trade_goods = naval_supplies
					trade_goods = tropical_wood
				}
			}
			custom_tooltip = MONOPOLY_INCOME_WOOD_DESC
		}
		custom_tooltip = estate_wolf_tengu_monopoly_of_wood_effect_tooltip
	}
	on_granted_province = {
		if = {
			limit = {
				OR = {
					trade_goods = naval_supplies
					trade_goods = tropical_wood
				}
			}
			add_province_modifier = {
				name = estate_wolf_tengu_monopoly_of_wood_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = estate_wolf_tengu_monopoly_of_wood_mod
			}
			remove_province_modifier = estate_wolf_tengu_monopoly_of_wood_mod
		}
	}
	on_revoked = {
		custom_tooltip = estate_wolf_tengu_monopoly_of_wood_revoke_tooltip
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = estate_wolf_tengu_monopoly_of_wood_mod
			}
			remove_province_modifier = estate_wolf_tengu_monopoly_of_wood_mod
		}
	}
	on_invalid = {
		custom_tooltip = estate_wolf_tengu_monopoly_of_wood_revoke_tooltip
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = estate_wolf_tengu_monopoly_of_wood_mod
			}
			remove_province_modifier = estate_wolf_tengu_monopoly_of_wood_mod
		}
	}
	penalties = {
	}
	benefits = {
	}
	cooldown_years = 10
	on_cooldown_expires = {
		country_event = { id = th_monopoly_privileges.8 }
	}
	ai_will_do = {
		factor = 0
	}
}
