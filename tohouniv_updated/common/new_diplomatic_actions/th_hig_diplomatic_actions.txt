th_lecture_ruler = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		has_government_attribute = enables_yama_authority
		FROM = { NOT = { has_country_modifier = th_lectured_modifier } }
		FROM = {
			capital_scope = {
				continent = gensokyo_continent
			}
		}
	}

	is_allowed = {
		if = {
			limit = {
				alliance_with = FROM
			}
			NOT = { alliance_with = FROM }
		}
		if = {
			limit = {
				has_opinion = { who = FROM value = 75 }
			}
			NOT = { has_opinion = { who = FROM value = 75 } }
		}
		if = {
			limit = {
				FROM = {
					is_at_war = no
					NOT = { had_recent_war = 10 }
				}
			}
			FROM = {
				OR = {
					had_recent_war = 10
					is_at_war = yes
				}
			}
		}
		if = {
			limit = {
				FROM = { is_in_war = { defenders = FROM } }
			}
			NOT = { FROM = { is_in_war = { defenders = FROM } } }
		}
		if = {
			limit = {
				war_with = FROM
			}
			NOT = { war_with = FROM }
		}
		if = {
			limit = {
				FROM = {
					has_country_flag = th_recently_lectured
					NOT = {
						had_country_flag = {
							flag = th_recently_lectured
							days = 7300
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_lecture_ruler_tt
				FROM = {
					OR = {
						NOT = { has_country_flag = th_recently_lectured }
						had_country_flag = {
							flag = th_recently_lectured
							days = 7300
						}
					}
				}
			}
		}
		if = {
			limit = {
				FROM = { has_country_modifier = th_grace_of_the_yama }
			}
			NOT = { FROM = { has_country_modifier = th_grace_of_the_yama } }
		}
		if = {
			limit = {
				truce_with = FROM
			}
			NOT = { truce_with = FROM }
		}
		if = {
			limit = {
				NOT = {
					estate_loyalty = {
						estate = estate_yama
						loyalty = 40
					}
				}
			}
			estate_loyalty = {
				estate = estate_yama
				loyalty = 40
			}
		}
		if = {
			limit = {
				NOT = { has_country_flag = th_improved_lecturing }
			}
			OR = {
				FROM = { NOT = { total_development = 51 } }
				AND = {
					FROM = { NOT = { total_development = 101 } }
					estate_influence = {
						estate = estate_yama
						influence = 15
					}
				}
				AND = {
					FROM = { NOT = { total_development = 201 } }
					estate_influence = {
						estate = estate_yama
						influence = 30
					}
				}
				AND = {
					FROM = { NOT = { total_development = 301 } }
					estate_influence = {
						estate = estate_yama
						influence = 45
					}
				}
				AND = {
					FROM = { NOT = { total_development = 401 } }
					estate_influence = {
						estate = estate_yama
						influence = 60
					}
				}
				estate_influence = {
					estate = estate_yama
					influence = 75
				}
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		FROM = {
			clr_country_flag = th_recently_lectured
			set_country_flag = th_recently_lectured
			country_event = {
				id = th_yama_authority_event.98
			}
		}
		add_estate_loyalty = {
			estate = estate_yama
			loyalty = -10
		}
	}

	ai_will_do = {
		always = no
	}
}

th_enforce_gensokyan_peace = {
	category = th_diplo_action

	require_acceptance = yes

	is_allowed = {
		custom_trigger_tooltip = {
			tooltip = th_enforce_gensokyan_peace_tt
			FROM = {
				any_known_country = {
					capital_scope = {
						continent = gensokyo_continent
					}
					is_in_war = {
						defender_leader = THIS
						attacker_leader = FROM
					}				
					NOT = { war_with = ROOT }
				}
			}
		}
		if = {
			limit = {
				war_with = FROM

			NOT = { war_with = FROM }
		}
		if = {
			limit = {
				is_at_war = yes
				NOT = { has_country_flag = th_improved_enforce_gensokyan_peace }
			}
			is_at_war = no
		}
		if = {
			limit = {
				truce_with = FROM
			}
			NOT = { truce_with = FROM }
		}
	}

		th_clear_copy_flag = yes
		random_country = {
			limit = {
				capital_scope = {
					continent = gensokyo_continent
				}

					defender_leader = THIS
					attacker_leader = FROM

			FROM = { white_peace = PREV }
			reverse_add_opinion = {
				modifier = th_enforced_peace_on_us

			}
		}
		FROM = {
			set_country_flag = th_hig_enforced_peace
		}
		country_event = {
			id = th_yama_authority_event.95
		}
		reverse_add_opinion = {
			modifier = th_enforced_peace_on_us
			who = FROM
		}
	}

	on_decline = {
		th_clear_copy_flag = yes
		FROM = {
			set_country_flag = th_hig_attacker

			limit = {
				capital_scope = {
					continent = gensokyo_continent
				}

					defender_leader = THIS
					attacker_leader = FROM

			}

			ROOT = { join_all_defensive_wars_of = PREV }

				modifier = th_enforced_peace_aid
				who = ROOT
			}
		}

				which = ai_value 
				value = 0.2
			}

				variable_name = from_manpower
				value = manpower
				who = FROM
			}
			subtract_variable = {
				which = ai_value
				which = from_manpower
			}
			export_to_variable = {
				variable_name = root_discipline
				value = modifier:discipline
				who = ROOT

							continent = gensokyo_continent

				tooltip = th_demand_unlawful_territory_ct

			limit = {
				NOT = {
					estate_loyalty = {
						estate = estate_yama
						loyalty = 1
					}
				}
			}
			estate_loyalty = {
				estate = estate_yama
				loyalty = 1
			}
		}
	}

	on_accept = {
		if = {
			limit = {
				has_country_flag = th_improved_demand_unlawful_territory
			}
			FROM = {
				th_yama_force_unlawful_territory_back = {
					improved_effect = yes
				}
			}
		}
		else = {
			add_estate_loyalty = {
				estate = estate_yama
				loyalty = -1
			}
			FROM = {
				th_yama_force_unlawful_territory_back = {}
			}
		}
		reverse_add_opinion = {
			who = FROM
			modifier = th_demanded_unlawful_territory
		}

		add_estate_loyalty = {
			estate = estate_yama
			loyalty = -1
		}
		FROM = {
			th_yama_refuse_unlawful_territory_back = yes
		}
		reverse_add_opinion = {
			who = FROM
			modifier = th_demanded_unlawful_territory
		}

			target = FROM

		}

			}

			}

			}
			export_to_variable = {
				variable_name = ai_value
				value = prestige
				who = ROOT
			}
			#-1 trust = -2 ai points
			multiply_variable = {
				which = ai_value
