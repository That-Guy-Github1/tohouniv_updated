#HIG diplo actions
th_lecture_ruler = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		has_country_flag = th_can_lecture_countries
	}

	is_allowed = {
		reverse_has_opinion_modifier = {
			who = FROM
			modifier = aggressive_expansion
		}
		NOT = {
			reverse_has_opinion = {
				who = FROM
				value = 0
			}
		}
		NOT = { alliance_with = FROM }
		FROM = { is_excommunicated = no }
		estate_influence = {
			estate = estate_yama
			influence = 5
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		excommunicate = FROM
		FROM = {
			add_ruler_modifier = {
				name = th_lectured_modifier
				duration = 3650
			}
			country_event = {
				id = th_yama_authority_event.99
				days = 3650
			}
		}
		add_estate_loyalty = {
			estate = estate_yama
			loyalty = 25
		}
		add_estate_influence_modifier = {
			estate = estate_yama
			desc = TH_YAMA_LECTURED_A_RULER
			influence = -10
			duration = 3650
		}
	}

	ai_will_do = {
		always = no
	}
}

th_enforce_gensokyan_peace = {
	category = th_diplo_action

	require_acceptance = yes

	is_visible = {
		has_country_flag = th_can_enforce_gensokyan_peace
	}

	is_allowed = {
		FROM = {
			any_known_country = {
				capital_scope = {
					continent = gensokyo_continent
				}
				is_in_war = {
					defender_leader = THIS
					attacker_leader = FROM
				}
			}
			NOT = { war_with = ROOT }
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		random_country = {
			limit = {
				capital_scope = {
					continent = gensokyo_continent
				}
				is_in_war = {
					defender_leader = THIS
					attacker_leader = FROM
				}
			}
			FROM = {
				white_peace = PREV
			}
		}
	}

	on_decline = {

	}

	ai_acceptance = {
        add_entry = {
            name = TH_DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
        }
	}
	ai_will_do = {
		always = no
	}
}

th_return_non_hell_provinces = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		has_government_attribute = enables_yama_authority
		FROM = {
			is_subject_of = ROOT
			any_owned_province = {
				continent = gensokyo_continent
				NOT = { region = former_hell_region }
				NOT = { region = hell_region }
				NOT = { region = netherworld_region }
				NOT = { area = Konngaras_state }
			}
			capital_scope = {
				NOT = { region = former_hell_region }
				NOT = { region = hell_region }
				NOT = { region = netherworld_region }
				NOT = { area = Konngaras_state }
			}
		}
	}

	is_allowed = {
		FROM = {
			NOT = { liberty_desire = 50 }
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		FROM = {
			th_return_gensokyan_provinces = yes
			add_country_modifier = {
				name = th_forced_to_release_provinces
				duration = 5475
			}
		}
	}

	ai_will_do = {
		always = no
	}
}
