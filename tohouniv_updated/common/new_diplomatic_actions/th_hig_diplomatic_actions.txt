#HIG diplo actions
th_lecture_ruler = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		has_country_flag = th_can_lecture_countries
	}

	is_allowed = {
		always = yes
	}

	on_accept = {
		th_clear_copy_flag = yes
		excommunicate = FROM
		#Remove excommunication effect after 10 years
		FROM = {
			add_ruler_modifier = {
				name = th_lectured_modifier
				duration = 3650
			}
		}
	}

	ai_will_do = {
		always = no
	}
}
