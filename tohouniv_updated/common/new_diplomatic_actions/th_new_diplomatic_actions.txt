#Create Tributary regardless of bordering the nation or not
th_create_tributary_state = {
	category = th_diplo_action
	alert_index = 30
	alert_tooltip = th_create_tributary_state_tooltip
	require_acceptance = yes
	is_visible = {
		has_government_attribute = has_tributaries
		FROM = {
			is_subject = no
			ai = yes
		}
	}
	is_allowed = {
		is_at_war = no
		FROM = {
			is_at_war = no
			is_great_power = no
			NOT = { num_of_subjects = 1 }
			custom_trigger_tooltip = {
				tooltip = th_create_tributary_state_dev_condition_tooltip
				AND = {
					NOT = { total_development = 200 }
					NOT = { total_development = ROOT }
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_create_tributary_state_strength_condition_tooltip
				NOT = {
					military_strength = {
						who = ROOT
						value = 0.3
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_create_tributary_state_distance_condition_tooltip
				NOT = {
					border_distance = {
						who = ROOT
						distance = 500
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_from_is_not_immune_to_diplomatic_actions_tt
				FROM = {
					NOT = { has_country_flag = th_immunity_to_diplomatic_actions }
					NOT = { has_government_attribute = enables_yama_authority }
				}
			}
		}
	}
	on_accept = {
		add_mandate = -1
		create_subject = {
			subject_type = tributary_state
			subject = FROM
		}
		FROM = {
			add_trust = {
				who = ROOT
				value = 5
			}
		}
	}
	on_decline = {
		add_prestige = -10
		add_trust = {
			who = FROM
			value = -10
			mutual = no
		}
	}
    ai_acceptance = {
        add_entry = {
            name = TH_DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
        }
		add_entry = {
			name = DIPLO_TRUST_WITH
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = TH_DIPLO_TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}
			#1 dev = -0.5
			multiply_variable = {
				which = ai_value
				value = -0.5
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_hostile
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000

		add_entry = {

		add_entry = {

					who = FROM

			}

				}
			}

				who = FROM
				attitude = attitude_protective
			}
			ai_attitude = {
				who = FROM
				attitude = attitude_domineering
			}
			FROM = {
				ai_attitude = {
					who = ROOT
					attitude = attitude_threatened
				}
			}
		}
	}
}

# Bribe Subject
th_bribe_subject = {
	category = th_diplo_action
	
	require_acceptance = no
	is_visible = {
		FROM = {
			is_subject = yes
			is_subject_of = ROOT
		}
	}
	is_allowed = {
		custom_trigger_tooltip = {
			tooltip = th_bribe_subjed_off_cd
			NOT = { has_country_modifier = th_bribed_subject_cd }
		}
		if = {
			limit = {
				FROM = {
					total_development = 1000
				}
			}
			years_of_income = 5.0
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 750
				}
			}
			years_of_income = 4
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 500
				}
			}
			years_of_income = 3
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 250
				}
			}
			years_of_income = 2
		}
		else = {
			years_of_income = 1
		}
		FROM = {
			NOT = {
				has_opinion = {
					who = ROOT
					value = 0
				}
			}
			has_opinion_modifier = {
				who = ROOT
				modifier = aggressive_expansion
			}
		}
		if = {
			limit = {
				FROM = {
					OR = {
						has_country_flag = th_immunity_to_diplomatic_actions
						has_government_attribute = enables_yama_authority
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_from_is_not_immune_to_diplomatic_actions_tt
				FROM = {
					NOT = { has_country_flag = th_immunity_to_diplomatic_actions }
					NOT = { has_government_attribute = enables_yama_authority }
				}
			}
		}
	}
	on_accept = {
		hidden_effect = {
			add_country_modifier = {
				name = th_bribed_subject_cd
				duration = 3650
				hidden = yes
			}
		}
		if = {
			limit = {
				FROM = {
					total_development = 1000
				}
			}
			add_years_of_income = -5
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 750
				}
			}
			add_years_of_income = -4
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 500
				}
			}
			add_years_of_income = -3
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 250
				}
			}
			add_years_of_income = -2
		}
		else = {
			add_years_of_income = -1
		}
		FROM = {
			add_treasury = 100
			add_opinion = {
				who = ROOT
				modifier = th_bribed_subject
			}
		}
	}

	ai_will_do = {
		if = {
			limit = {
				FROM = {
					total_development = 1000
				}
			}
			years_of_income = 5.0
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 750
				}
			}
			years_of_income = 4
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 500
				}
			}
			years_of_income = 3
		}
		else_if = {
			limit = {
				FROM = {
					total_development = 250
				}
			}
			years_of_income = 2
		}
		else = {
			years_of_income = 1
		}
		FROM = {
			NOT = {
				has_opinion = {
					who = ROOT
					value = 0
				}
			}
			has_opinion_modifier = {
				who = ROOT
				modifier = aggressive_expansion
			}
		}
	}
}

# Break vassal alliances
th_break_vassal_alliance = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		FROM = {
			is_subject_of = ROOT
			OR = {
				is_subject_of_type = th_yukari_daimyo 
				is_subject_of_type = th_special_march 
				is_subject_of_type = th_yukari_puppet_state
				is_subject_of_type = th_pet_state
			}
		}
	}

	is_allowed = {
		FROM = {
			num_of_allies = 1
			any_ally = {
				is_subject = no
			}
		}
	}

	on_accept = {
		FROM = {
			every_ally = {
				limit = {
					is_subject = no
				}
				add_opinion = {
					who = FROM
					modifier = th_forced_alliance_break
					years = 5
				}
				reverse_add_opinion = {
					who = FROM
					modifier = th_forced_alliance_break
					years = 5
				}
				add_opinion = {
					who = FROM
					modifier = th_forced_alliance_break_2
					years = 5
				}
				reverse_add_opinion = {
					who = FROM
					modifier = th_forced_alliance_break_2
					years = 5
				}
			}
			add_opinion = {
				who = ROOT
				modifier = th_overlord_forced_alliance_break				
			}
			add_country_modifier = {
				name = th_alliance_breaker
				duration = 365
			}
		}
	}
}

# Elevate Government Rank
th_elevate_subject_government_rank = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		FROM = {
			is_subject_of = ROOT
			OR = {
				is_subject_of_type = personal_union
				is_subject_of_type = th_yukari_daimyo
				is_subject_of_type = th_yukari_puppet_state
				is_subject_of_type = th_yukari_march
				is_subject_of_type = th_special_march 
				is_subject_of_type = th_yuuka_gatekeeper
				is_subject_of_type = th_pet_state
				is_subject_of_type = th_yama_executor
			}
		}
		OR = {
			AND = {
				government_rank = 2
				FROM = { NOT = { government_rank = 2 } }
			}
			AND = {
				government_rank = 3
				FROM = { NOT = { government_rank = 3 } }
			}
		}
	}

	is_allowed = {
		OR = {
			AND = {
				government_rank = 2
				FROM = { NOT = { government_rank = 2 } }
				prestige = 50
			}
			AND = {
				government_rank = 3
				FROM = { NOT = { government_rank = 3 } }
				prestige = 75
			}
		}
	}

	on_accept = {
		if = {
			limit = {
				government_rank = 3
			}
			FROM = {
				set_government_rank = 3
			}
		}
		else = {
			FROM = {
				set_government_rank = 2
			}
		}
	}
}

#Steal target's treasury - diplo action, which can be assigned to another tag later on
th_loot_from_treasury = {
	category = th_diplo_action
	require_acceptance = no

	is_visible = {
		has_country_flag = th_enable_burgling_action
	}

	is_allowed = {
		has_spy_network_in = {
			who = FROM
			value = 50
		}
		custom_trigger_tooltip = {
			tooltip = th_diplomacy_has_already_looted_treasury
			FROM = {				
				OR = {
					NOT = { has_country_flag = th_has_been_already_looted_by_spies_of_@ROOT }
					had_country_flag = {
						flag = th_has_been_already_looted_by_spies_of_@ROOT
						days = 9126
					}
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = th_diplomacy_cant_loot_from_allies
			NOT = { alliance_with = FROM }
		}
		custom_trigger_tooltip = {
			tooltip = th_diplomacy_has_not_truce_with_target
			NOT = { truce_with = FROM }
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		FROM = {
			clr_country_flag = th_has_been_already_looted_by_spies_of_@ROOT
			country_event = {
				id = th_diplo_events.14
			}
		}
		add_spy_network_in = {
			who = FROM
			value = -50			
		}
	}

	ai_will_do = {
		always = no
	}
}
