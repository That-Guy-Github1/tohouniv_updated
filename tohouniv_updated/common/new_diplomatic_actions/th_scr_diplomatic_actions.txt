#SCR demand war support from another country
th_demand_war_support = {
	category = th_diplo_action
	alert_index = 30
	alert_tooltip = th_request_conversion_tooltip
	require_acceptance = yes

	is_visible = {
		OR = {
			th_has_copied_from_target = {
				target = SCR
			}
			has_country_flag = th_scarlet_charisma_enabled
		}
		FROM = {
			is_free_or_tributary_trigger = yes
			alliance_with = ROOT
		}
	}

	is_allowed = {
		custom_trigger_tooltip = {
			tooltip = th_demand_war_support_has_been_requested_tt
			FROM = {				
				OR = {
					NOT = { has_country_flag = th_was_already_requested_for_soldiers_@ROOT }
					had_country_flag = {
						flag = th_was_already_requested_for_soldiers_@ROOT
						days = 1825
					}
				}
			}
		}
		if = {
			limit = {
				FROM = { is_at_war = yes }
			}
			custom_trigger_tooltip = {
				tooltip = TH_IS_AT_WAR_TT
				FROM = { is_at_war = no }
			}
		}
		custom_trigger_tooltip = {
			tooltip = th_we_are_not_at_war_with_from_allies_tt
			NOT = {
				any_country = {
					war_with = ROOT
					alliance_with = FROM
				}
			}
		}		
		custom_trigger_tooltip = {
			tooltip = th_from_has_at_least_70_trust_tt
			FROM = {
				trust = { who = ROOT value = 70 }
			}
		}
		custom_trigger_tooltip = {
			tooltip = th_from_has_at_least_thousand_man
			FROM = {
				manpower = 1
			}
		}
		if = {
			limit = {
				FROM = {
					OR = {
						has_country_flag = th_immunity_to_diplomatic_actions
						has_government_attribute = enables_yama_authority
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_from_is_not_immune_to_diplomatic_actions_tt
				FROM = {
					NOT = { has_country_flag = th_immunity_to_diplomatic_actions }
					NOT = { has_government_attribute = enables_yama_authority }
				}
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		export_to_variable = {
			which = th_manpower_to_conscript
			value = manpower
			who = FROM
		}
		divide_variable = {
			which = th_manpower_to_conscript
			value = 10	#10% of manpower gets transferred to SCR
		}
		while = {
			limit = {
				check_variable = {
					which = th_manpower_to_conscript
					value = 0.1
				}
			}
			add_manpower = 0.1
			FROM = {
				add_manpower = -0.1
			}
			subtract_variable = {
				which = th_manpower_to_conscript
				value = 0.1
			}
		}
		if = {
			limit = {
				has_country_flag = th_scarlet_charisma_upgraded
			}
			FROM = {
				add_trust = {
					who = ROOT
					value = -5
				}
			}
		}
		else = {
			FROM = {
				add_trust = {
					who = ROOT
					value = -10
				}
			}
		}
		FROM = {
			clr_country_flag = th_was_already_requested_for_soldiers_@ROOT
			set_country_flag = th_was_already_requested_for_soldiers_@ROOT
			set_country_flag = scr_has_been_target_of_scarlet_charimsa
		}
	}

	on_decline = {
	}

    ai_acceptance = {
        add_entry = {
            name = TH_DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
        }
		add_entry = {
			name = TH_DIPLO_TRUST_WITH
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_hostile
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_outraged
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000
			}
		}
		add_entry = {
			name = DIPLO_IS_RIVAL
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_rivalry
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000
			}
		}
		add_entry = {
			name = DIPLO_IS_THREAT
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_threatened
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -100
			}
		}
		add_entry = {
			name = DIPLO_DIPLOMATIC_REPUTATION
			limit = { }
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 2
			multiply_variable = {
				which = ai_value
				value = 2
			}
		}
		add_entry = {
			name = DIPLO_HISTORICAL_RIVAL
			
			limit = {
				OR = { 
					historical_rival_with = FROM
					FROM = { historical_rival_with = ROOT }
				}
			}
			
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -100
			}
		}
		add_entry = {
			name = TH_BASE_DISOBEDIENCE
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -25
			}
		}
		add_entry = {
			name = TH_NEGATIVE_PRESTIGE
			
			limit = {
				NOT = { prestige = 0 }
			}
			export_to_variable = {
				variable_name = ai_value
				value = prestige
				who = ROOT
			}
			#-1 trust = -2 ai points
			multiply_variable = {
				which = ai_value
				value = 0.2
			}
		}
    }
	ai_will_do = {
		always = no
	}
}

#SCR trade off prestige against trust with the target
th_scarlet_charisma = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		OR = {
			th_has_copied_from_target = {
				target = SCR
			}
			has_country_flag = th_scarlet_charisma_enabled
		}
	}

	is_allowed = {
		if = {
			limit = {
				has_country_flag = th_scarlet_charisma_upgraded
			}
			prestige = 10
		}
		else = {
			prestige = 20
		}
		FROM = {
			NOT = { is_rival = ROOT }
			NOT = { is_enemy = ROOT }
			NOT = { war_with = ROOT }
		}
		if = {
			limit = {
				FROM = {
					OR = {
						has_country_flag = th_immunity_to_diplomatic_actions
						has_government_attribute = enables_yama_authority
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_from_is_not_immune_to_diplomatic_actions_tt
				FROM = {
					NOT = { has_country_flag = th_immunity_to_diplomatic_actions }
					NOT = { has_government_attribute = enables_yama_authority }
				}
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		FROM = {
			add_trust = {
				who = ROOT
				value = 5
			}
		}
		if = {
			limit = {
				has_country_flag = th_scarlet_charisma_upgraded
			}
			add_prestige = -10
		}
		else = {
			add_prestige = -20
		}
	}
	
	ai_will_do = {
		always = no
	}
}
