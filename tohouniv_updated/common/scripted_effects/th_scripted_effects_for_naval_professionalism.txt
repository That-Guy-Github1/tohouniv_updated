add_naval_professionalism_effect = {
	if = {
		limit = { has_government_attribute = th_has_naval_professionalism }
		hidden_effect = {
			clear_naval_professionalism_buffs = yes
		}
		change_variable = {
			naval_professionalism_var = $value$
		}
		custom_tooltip = TH_CURRENT_NAVAL_PROFESSIONALISM_TOOLTIP
		hidden_effect = {
			if = {
				limit = { check_variable = { naval_professionalism_var = 100 } }
				set_variable = { naval_professionalism_var = 100 }
			}
			if = {
				limit = { NOT = { check_variable = { naval_professionalism_var = 0 } } }
				set_variable = { naval_professionalism_var = 0 }
			}
			add_naval_professionalism_buffs = yes
		}
	}
}

th_add_yearly_naval_professionalism_effect = {
	if = {
		limit = { has_government_attribute = th_has_naval_professionalism }
		hidden_effect = {
			export_to_variable = {
				which = yearly_naval_professionalism_variable
				value = modifier:yearly_naval_professionalism
			}
			multiply_variable = {
				which = yearly_naval_professionalism_variable
				value = 100
			}
			clear_naval_professionalism_buffs = yes
		}
		change_variable = {
			which = naval_professionalism_var
			which = yearly_naval_professionalism_variable
		}
		custom_tooltip = TH_CURRENT_NAVAL_PROFESSIONALISM_TOOLTIP
		hidden_effect = {
			if = {
				limit = { check_variable = { naval_professionalism_var = 100 } }
				set_variable = { naval_professionalism_var = 100 }
			}
			if = {
				limit = { NOT = { check_variable = { naval_professionalism_var = 0 } } }
				set_variable = { naval_professionalism_var = 0 }
			}
			add_naval_professionalism_buffs = yes
		}
	}
}
