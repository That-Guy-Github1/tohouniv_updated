###PROVINCE HAS FORT BUILDING###
#Province has any fort building
th_has_fort_building = {
	OR = {
		has_building = fort_15th
		has_building = fort_16th
		has_building = fort_17th
		has_building = fort_18th
	}
}
th_has_outdated_fort_building = {
	if = {
		limit = {
			$tag$ = {
				mil_tech = 24
			}
		}
		NOT = { has_building = fort_18th }
		OR = {
			has_building = fort_17th
			has_building = fort_16th
			has_building = fort_15th
		}
	}
	else_if = {
		limit = {
			$tag$ = {
				mil_tech = 19
			}
		}
		NOT = { has_building = fort_17th }
		OR = {
			has_building = fort_16th
			has_building = fort_15th
		}
	}
	else_if = {
		limit = {
			$tag$ = {
				mil_tech = 14
			}
		}
		NOT = { has_building = fort_16th }
		has_building = fort_15th
	}
}

#Province has an up-to-date fort building. As mil tech reference will be the tag used
th_has_up_to_date_fort_building = {
	if = {
		limit = {
			$tag$ = {
				mil_tech = 24
			}
		}
		has_building = fort_18th
	}
	else_if = {
		limit = {
			$tag$ = {
				mil_tech = 19
			}
		}
		has_building = fort_17th
	}
	else_if = {
		limit = {
			$tag$ = {
				mil_tech = 14
			}
		}
		has_building = fort_16th
	}
	else = {
		has_building = fort_15th
	}
}

#For more colony types
th_num_of_colonies = {
	if = {
		limit = {
			has_dlc = "Leviathan"
		}
		calc_true_if = {
			all_subject_country = {
				OR = {
					is_subject_of_type = crown_colony
					is_subject_of_type = private_enterprise
					is_subject_of_type = self_governing_colony
				}
			}
			amount = $amount$
		}
	}
	else = {
		calc_true_if = {
			all_subject_country = {
				is_subject_of_type = colony
			}
			amount = $amount$
		}
	}
}

th_is_colony = {
	if = {
		limit = {
			has_dlc = "Leviathan"
		}
		OR = {
			is_subject_of_type = crown_colony
			is_subject_of_type = private_enterprise
			is_subject_of_type = self_governing_colony
		}
	}
	else = {
		is_subject_of_type = colony
	}
}
