#Hire Youmu as an advisor
country_event = {
	id = th_advisor_events.1
	title = th_advisor_events.1.t
	desc = th_advisor_events.1.desc
	picture = YOUMU_eventPicture
	
	is_triggered_only = yes

	hidden = yes

	immediate = {
		hidden_effect = {
			set_country_flag = th_hired_advisor_youmu
		}
	}

	option = {
		name = th_advisor_events.1.a
		define_advisor = {
			type = th_advisor_youmu
			name = "Youmu Konpaku"
			skill = 3
			location = 4958
			discount = yes
			female = yes
			culture = Netherlandish
			religion = ROOT
		}
		hidden_effect = {
			country_event = {  		#Clear event after 30 years
				id = th_advisor_events.2
				days = 18250
			}
		}

		ai_chance = { 
			factor = 100
		}
	}
}
