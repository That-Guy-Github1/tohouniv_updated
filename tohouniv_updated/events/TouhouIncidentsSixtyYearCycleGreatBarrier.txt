				NOT = { has_province_modifier = th_POFV_random_flower_bed }

				name = th_POFV_random_flower_bed
				duration = -1
			}
		}
	}

		random_country = {
			limit = { 
				th_incident_can_become_investigator = yes
			}

				id = th_sixty_year_cycle_incident_events.1
			}
		}

			factor = 100
		}
	}

country_event = {
	id = th_sixty_year_cycle_incident_events.3
	title = none
	desc = none
	picture = HIGAN_eventPicture
	hidden = yes
	
	trigger = {
		th_current_incident = { incident = sixty_year_cycle_incident }
		has_global_flag = th_POFV_random_matches_active
		owns = 1
	}

	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		if = {
			limit = {
				1 = { NOT = { has_province_flag = th_POFV_counter_active } }
			}
			th_set_incident_stage = {
				stage = 0
			}
			1 = { set_province_flag = th_POFV_counter_active }
		}
		else = {
			th_prepare_for_next_stage = yes
		}
		if = {
			limit = {
				th_incident_has_reached_stage = {
					stage = 5
				}
			}
			random_country = {
				limit = {
					tag = HIG
					#TODO count highest score
				}
				country_event = {
					id = th_sixty_year_cycle_incident_events.7
				}
			}
		}
		every_country = {
			limit = {
				has_country_flag = th_POFV_joined_the_investigation
			}
			clr_country_flag = th_POFV_search_for_enemy
			set_country_flag = th_POFV_search_for_enemy
			1 = { set_province_flag = th_POFV_enemies_set }
		}
		if = {
			limit = {
				1 = { has_province_flag = th_POFV_enemies_set }
			}
			random_country = {
				limit = {
					has_country_flag = th_POFV_search_for_enemy
				}
				country_event = {
					id = th_sixty_year_cycle_incident_events.4
				}
			}
			1 = { clr_province_flag = th_POFV_enemies_set }
		}
		
		ai_chance = { 
			factor = 100
		}
	}
}
