country_event = {  ### Select what kind of weather you want to use on yourself
	id = flavor_hvr.101
	title = flavor_hvr.101.t
	picture = NITORI_GADGET_eventPicture
	desc = flavor_hvr.101.desc
	is_triggered_only = yes

	
	option = { #Use heavy rain
		name = flavor_hvr.101.a
		if = {
			limit = {
				NOT = { has_country_modifier = hvr_heavy_rain}
			}
			add_country_modifier = {
				name = hvr_heavy_rain
				duration = 1850
			}
			add_adm_power = -50
		}
		else = { custom_tooltip = flavor_hvr.101.a.ct }	
	}

	option = { #Use terraforming
		name = flavor_hvr.101.b
		if = {
			limit = {
				NOT = { has_country_modifier = hvr_terraforming }
			}
			add_country_modifier = {
				name = hvr_terraforming 
				duration = 1850
			}
			random_owned_province = {
				limit = {
					is_state_core = root
					area_for_scope_province = {
						is_capital_of = root
					}
				}
				random_list = {
					33 = { add_base_tax = 1 } 
					33 = { add_base_manpower = 1 } 
					33 = { add_base_production = 1 } 
				}
			}
			add_adm_power = -50
		}
		else = { custom_tooltip = flavor_hvr.101.b.ct }		
	}
	
	option = { # Experiments with electricity
		name = flavor_hvr.101.c
		if = {
			limit = {
				NOT = { has_country_modifier = hvr_experiment_with_electricity }
			}
			add_country_modifier = {
				name = hvr_experiment_with_electricity
				duration = 1850
			}
			random_list = {
				33 = { change_innovativeness = 1 } 
				33 = { change_innovativeness = 3 } 
				33 = { change_innovativeness = 5 } 
			}
			add_adm_power = -25
			add_dip_power = -25
		}
		else = { custom_tooltip = flavor_hvr.101.c.ct }		
	}

	option = { # Create Rainbow!
	name = flavor_hvr.101.d
	if = {
		limit = {
			NOT = { has_country_modifier = hvr_rainbow }
		}
		add_country_modifier = {
			name = hvr_rainbow
			duration = 2555 #7 years
		}
		add_adm_power = -25
		add_dip_power = -25
		add_mil_power = -25
	}
	else = { custom_tooltip = flavor_hvr.101.d.ct }		
}

	option = {  #Take me back
		name = flavor_hvr.101.e
		ai_chance = {
			factor = 50
		}
	}
}
