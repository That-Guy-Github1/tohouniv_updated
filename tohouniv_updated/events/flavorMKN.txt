namespace = flavor_mkn

#PU Shinki
country_event = {
	id = flavor_mkn.1
	title = flavor_mkn.1.t
	desc = flavor_mkn.1.desc
	picture = SHINKI_eventPicture
    
    is_triggered_only = yes
	
	option = {	#We need to persuade Shinki
		name = flavor_mkn.1.a
		trigger = {		
			MKI = {
				is_free_or_tributary_trigger = yes
				has_ruler = "Shinki"
				dynasty = "of Makai"
			}
		}
		highlight = yes
		add_years_of_income = -5		
		create_subject = {
			subject_type = personal_union
			subject = MKI
		}
		define_ruler = {
			name = "Shinki"
			dynasty = "of Makai"
			culture = Makaian
			adm = 6
			dip = 6
			mil = 6
			age = 20
			claim = 100
			female = yes
		}
		remove_heir = { allow_new_heir = no }
		hidden_effect = {
			add_ruler_personality = immortal_personality
			set_ruler_flag = touhou_character_ruler
			set_ruler_flag = touhou_original_ruler
		}
		ai_chance = { 
			factor = 0
		}
	}
	
	option = {	#We need to subjugate Shinki!
		name = flavor_mkn.1.b
		trigger = {
			MKI = {
				ai = yes
			}
		}
		declare_war_with_cb = {
			who = MKI
			casus_belli = cb_restore_personal_union
		}
		
		ai_chance = { 
			factor = 0
		}
	}
	
	option = {	#That's just ridiculous.
		name = flavor_mkn.1.c
		hidden_effect = {	
			add_casus_belli = {
				target = MKI
				type = cb_restore_personal_union
				months = 900
			}
		}		

		ai_chance = { 
			factor = 100
		}
	}
}

#Gold of Siberia
country_event = {
	id = flavor_mkn.3
	title = flavor_mkn.3.t
	desc = flavor_mkn.3.desc
	picture = GOLDEN_CITY_eventPicture
	fire_only_once = yes

	trigger = {
		th_is_or_was_tag = {
			tag = MKN
		}
		num_of_owned_provinces_with = {
			region = east_siberia_region
			value = 15
		}
		NOT = {
			num_of_owned_provinces_with = {
				region = east_siberia_region
				trade_goods = gold
				value = 6
			}
		}
	}

	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					region = east_siberia_region
					NOT = { trade_goods = gold }
					NOT = { has_province_flag = mkn_gold_flag }
				}
				set_province_flag = mkn_gold_flag
			}
			random_owned_province = {
				limit = {
					region = east_siberia_region
					NOT = { trade_goods = gold }
					NOT = { has_province_flag = mkn_gold_flag }
				}
				set_province_flag = mkn_gold_flag
			}
			random_owned_province = {
				limit = {
					region = east_siberia_region
					NOT = { trade_goods = gold }
					NOT = { has_province_flag = mkn_gold_flag }
				}
				set_province_flag = mkn_gold_flag
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	option = {	#GOLD!
		name = flavor_mkn.3.a
		every_owned_province = {
			limit = {
				has_province_flag = mkn_gold_flag
			}
			change_trade_goods = gold
			hidden_effect = {
				clr_province_flag = mkn_gold_flag
			}
		}
		ai_chance = { 
			factor = 100
		}
	}
}
