#Visit of Suwako Moriya in Higan

	title = flavor_mry.1.t
	desc = flavor_mry.1.desc
	picture = SUWAKO_eventPicture

	major = yes

			limit = {

				}
			}
		}

			every_heretic_province = {
				limit = {
					region = hell_region

				}
			}
		}
		
		ai_chance = { 
			factor = 100
		}
	}
	
	option = {	#Not worth it
		name = flavor_mry.1.b
		
		add_adm_power = 100
		
		ai_chance = { 
			factor = 0
		}
	}
}

#Automatically convert province with MRY's claim
country_event = {
	id = flavor_mry.101
	title = "flavor_mry.101.t"
	desc = "flavor_mry.101.desc"
	picture = SCARLETT_eventPicture
	
	trigger = {
		any_province = {
			NOT = { is_permanent_claim = ROOT }
			is_claim = ROOT
			NOT = { religion = ROOT }
			is_reformation_center = no
			NOT = {	has_province_modifier = religious_zeal_at_conv }
			NOT = {	has_active_triggered_province_modifier = th_religious_center }
			NOT = {	has_province_modifier = religious_center }
			is_empty = no
			has_missionary = no
			NOT = { has_state_edict = edict_religion_enforced }
			NOT = { has_province_flag = th_has_been_converted }
			owner = {
				NOT = { is_subject_of = ROOT }
			}
		}
		OR = {
			AND = {
				has_country_flag = th_enabled_hakurei_converting_on_claim
				religion = th_hakureishinto
			}
			AND = {
				has_country_flag = th_enabled_taoist_converting_on_claim
				religion = th_taoist
			}
			AND = {
				has_country_flag = th_enabled_buddhist_converting_on_claim
				religion = th_buddhist
			}
			AND = {
				has_country_flag = th_enabled_moriya_converting_on_claim
				religion = th_moriyashinto
			}
			AND = {
				has_country_flag = th_enabled_oldhakurei_converting_on_claim
				religion = th_oldhakurei
			}
			AND = {
				has_country_flag = th_enabled_shinkism_converting_on_claim
				religion = th_shinkism
			}
			AND = {
				has_country_flag = th_enabled_christian_converting_on_claim
				religion = th_christian
			}
			AND = {
				has_country_flag = th_enabled_lunarian_converting_on_claim
				religion = th_lunarian_religion
			}
			has_country_flag = th_enabled_converting_on_claim
		}
		OR = {
			AND = {
				has_reform = Gensokyan_shrine_theocracy
				has_country_flag = th_enabled_converting_on_claims_for_shrines
			}
			has_country_flag = th_enabled_converting_on_claims_without_restriction
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	hidden = yes
	
	option = {		#Convert people here to the only one true faith!
		name = "flavor_mry.101.a"
		every_province = {
			limit = {
				NOT = { is_permanent_claim = ROOT }
				is_claim = ROOT
				NOT = { religion = ROOT }
				is_reformation_center = no
				NOT = {	has_province_modifier = religious_zeal_at_conv }
				NOT = {	has_active_triggered_province_modifier = th_religious_center }
				NOT = {	has_province_modifier = religious_center }
				is_empty = no
				has_missionary = no
				NOT = { has_state_edict = edict_religion_enforced }
				NOT = { has_province_flag = th_has_been_converted }
				owner = {
					NOT = { is_subject_of = ROOT }
				}
			}
			change_religion = ROOT
			set_province_flag = th_has_been_converted
		}

		ai_chance = { 
			factor = 100
		}
	}
}

#lose every religious tributary if MRY changed tag or changed government or changed religion
country_event = {
	id = flavor_mry.103
	title = flavor_mry.103.t
	desc = flavor_mry.103.desc
	picture = KANAKOBOSS_eventPicture
	
	mean_time_to_happen = {
		months = 1
	}
	
	trigger = {
		any_subject_country = { is_subject_of_type = th_religious_tributary }
		OR = {
			AND = {
				NOT = { tag = MRY }
				NOT = { tag = GNK }
			}
			NOT = {	has_reform = Gensokyan_shrine_theocracy }
			NOT = { religion = th_moriyashinto }
		}

		num_of_subjects = 1 
	}
	
	immediate = {
		hidden_effect = {
			every_subject_country = { 
				limit = { is_subject_of_type = th_religious_tributary } 
				country_event = { id = flavor_mry.104 }
			}
		}
	}
	
	option = {	#Well, we shouldn`t abandon our old tradition.
		name = flavor_mry.103.a
		custom_tooltip = flavor_mry.103.a.tt
	}
}

#Moriya`s tributaries get independent

	desc = flavor_mry.104.desc
	picture = LIBERUM_VETO_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_subject = yes
		is_subject_of_type = th_religious_tributary
		overlord = {
			OR = {
				AND = {
					NOT = { tag = MRY }
					NOT = { tag = GNK }
				}
				NOT = {	has_reform = Gensokyan_shrine_theocracy }
				NOT = { religion = th_moriyashinto }
			}
		}
	}
	
	option = {	#We should not follow these evil gods any longer!
		name = flavor_mry.104.a
		grant_independence = yes
	}
}
