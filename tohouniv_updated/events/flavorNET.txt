#PU Yukari
country_event = {
	id = flavor_net.1
	title = flavor_net.1.t
	desc = flavor_net.1.desc
	picture = YUKARI2_eventPicture
    
    is_triggered_only = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = {
                    has_ruler = "Yuyuko"
                    dynasty = "Saigyouji"
				}
				th_getruler_diplomatic_skill = {
					SKILL_BONI = 600
				}
			}
			else = {
				th_getruler_diplomatic_skill = {
					SKILL_BONI = 0
				}
			}
			#Yuyukos base diplo fight skills
			set_variable = {
				which = th_yukari_diplo_skills
				value = 1250
			}
			th_defineopponent_fighting_skill = {
				VARIABLE = th_yukari_diplo_skills
			}
			if = {
				limit = {
					NOT = { has_global_flag = th_deactivated_incidents }
				}
				country_event = {
					id = th_danmaku_event.20
				}
			}
		}
	}
	
	option = {	#Yukari is no longer capable of ruling her realm! I have to fight her!
		name = flavor_net.1.a
		trigger = {
			NOT = { has_global_flag = th_deactivated_incidents }
			MHG = {
				was_player = no
				ai = yes
			}
		}
		highlight = yes
		custom_tooltip = flavor_net.1.a.tt
		th_danmaku_diplo_fight_ruler_and_opponent = {
			win_event = flavor_net.2
			draw_event = flavor_net.2
			defeat_event = flavor_net.3
			days = 20
		}
		
		ai_chance = { 
			factor = 0
		}
	}
	
	option = {	#We have to take over the throne!
		name = flavor_net.1.b
		trigger = {
			MHG = {
				was_player = no
				ai = yes
			}
		}
		custom_tooltip = flavor_net.1.b.tt
		declare_war_with_cb = {
			who = MHG
			casus_belli = cb_restore_personal_union
		}
		
		ai_chance = { 
			factor = 0
		}
	}
	
	option = {	#We have a claim on Yuyukos throne
		name = flavor_net.1.c
		custom_tooltip = flavor_net.1.c.tt
		add_casus_belli = {
			target = MHG
			type = cb_restore_personal_union
			months = 900
		}
		
		ai_chance = { 
			factor = 0
		}
	}
	
	option = {	#I trust in Yuyuko's judgement
		name = flavor_net.1.d
		custom_tooltip = flavor_net.1.d.tt
		add_opinion = {
			who = MHG
			modifier = mhg_high_trust
		}
		reverse_add_opinion = {
			who = MHG
			modifier = mhg_high_trust
		}
		add_trust = {
			who = MHG
			value = 25
			mutual = yes
		}
		
		ai_chance = { 
			factor = 100
		}
	}
}

#Necromancer attacks Yuyuko
country_event = {
	id = flavor_net.4
	title = flavor_net.4.t
	desc = flavor_net.4.desc
	picture = YUYUKO_eventPicture
	
	trigger = {
		th_is_or_was_tag = {
			tag = NET
		}
		has_ruler_flag = th_original_yuyuko
		has_mission = NET_elimiate_seiga
		NOT = { mission_completed = NET_elimiate_seiga }
		OR = {
			is_year = 1500
			mission_completed = NET_recover_mystias_cart
		}
	}

	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.1
			exists = SNK
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = net_ambushed_by_seiga
			th_getruler_fighting_skill = {
				SKILL_BONI = 0
			}
			set_variable = {
				which = net_seigas_ambush_combat_skills
				value = 250
			}
			th_defineopponent_fighting_skill = {
				VARIABLE = net_seigas_ambush_combat_skills
			}
		}
	}
	
	option = {	#I misjudged her. She can still hold responsibilty apparently
		name = flavor_net.4.a
		custom_tooltip = flavor_net.4.a.tt
		highlight = yes
		th_danmaku_normal_fight_ruler_and_opponent = {
			win_event = flavor_net.5
			draw_event = flavor_net.5
			defeat_event = flavor_net.5
			days = 3
		}
		
		ai_chance = { 
			factor = 100
		}
	}
}

	is_triggered_only = yes

			limit = {
				NOT = {	is_core = ROOT }
				NOT = { is_permanent_claim = ROOT }
			}

		}

		}

			swap_non_generic_missions = yes
		}
		
		ai_chance = { 
			factor = 100
		}
	}
}

#Pick Mortal Policy
country_event = {
	id = flavor_net.100
	title = flavor_net.100.t
	desc = flavor_net.100.desc
	picture = YUYUKO_eventPicture
	
	is_triggered_only = yes
	
	option = {	#Acceptance
		name = flavor_net.100.a

			limit = {
				OR = {
