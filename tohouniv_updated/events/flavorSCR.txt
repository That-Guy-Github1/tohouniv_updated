	id = flavor_scr.2
	title = flavor_scr.2.t
	desc = flavor_scr.2.desc
	picture = THEATRE_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = SCR
			AND = {
				OR = {
					tag = SCD
					tag = GNK
				}
				OR = {
					primary_culture = Scarlett
					dynasty = "Scarlet"
				}
			}
		}
		is_year = 1790
		NOT = { is_year = 1827 }
	}
	
	mean_time_to_happen = {
		months = 180
		
		modifier = {
			factor = 0.75
			exists = HAB
		}
		modifier = {
			factor = 0.25
			has_ruler_flag = th_original_remi
		}
	}
	
	option = {	#Hire this man! I want him to compose a theme for me!
		name = flavor_scr.2.a
		add_years_of_income = -0.25
		add_country_modifier = { 
			name = th_scr_septette_of_a_dead_princess
			duration = -1
		}
		if = {
			limit = {
				exists = HAB
			}
			HAB = {
				add_opinion = {
					who = ROOT
					modifier = th_admires_our_culture
				}
			}
		}

		add_prestige = 10
		
		ai_chance = { 
			factor = 50 
			modifier = {
				factor = 100

			}
		}
	}
	
	option = {	#We need the money for more important stuff.
		name = flavor_scr.2.b
